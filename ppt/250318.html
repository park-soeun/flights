def remove_outliers(df, column):
  Q1 = df[column].quantile(0.25)  # 1ì‚¬ë¶„ìœ„ìˆ˜ (Q1)
  Q3 = df[column].quantile(0.75)  # 3ì‚¬ë¶„ìœ„ìˆ˜ (Q3)
